Analog forecasting

using Plots, DifferentialEquations, AnalogDataAssimilation


σ = 10.0
ρ = 28.0
β = 8.0 / 3

dt_integration = 0.01
dt_states = 1
dt_obs = 8
var_obs = [1]
nb_loop_train = 100
nb_loop_test = 10
sigma2_catalog = 0.0
sigma2_obs = 2.0

ssm = StateSpaceModel(
    lorenz63,
    dt_integration,
    dt_states,
    dt_obs,
    [σ, ρ, β],
    var_obs,
    nb_loop_train,
    nb_loop_test,
    sigma2_catalog,
    sigma2_obs,
)

u0 = [8.0; 0.0; 30.0]

prob = ODEProblem(ssm.model, u0, tspan, ssm.params)
u0 = last(solve(prob, reltol = 1e-6, save_everystep = false))
